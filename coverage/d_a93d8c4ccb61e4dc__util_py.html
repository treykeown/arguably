39def split_unquoted(unsplit: str, delimeter: str, limit: Union[int, float] = math.inf) -> List[str]: 

40 """Splits text at a delimiter, as long as that delimiter is not quoted (either single ' or double quotes ").""" 

41 assert len(delimeter) == 1 

42 assert limit > 0 

43 result = list() 

44 quote_char = None 

45 accumulator: List[str] = list() 

46 for char in unsplit: 

47 if char == delimeter and quote_char is None and limit > 0: 

48 result.append("".join(accumulator)) 

49 accumulator.clear() 

50 limit -= 1 

51 continue 

52 elif char == "'": 

53 if quote_char is None: 

54 quote_char = "'" 

55 elif quote_char == "'": 

56 quote_char = None 

57 elif char == '"': 

58 if quote_char is None: 

59 quote_char = '"' 

60 elif quote_char == '"': 

61 quote_char = None 

62 accumulator.append(char) 

63 result.append("".join(accumulator)) 

64 accumulator.clear() 

65 return result

79def find_alias(used_aliases: Collection[str], name: str) -> Optional[str]: 

80 """ 

81 Simple algorithm for automatically finding an alias for a parameter. There are better ways, but this works for now. 

82 Iterates over each character in a parameter, tries both lowercase and uppercase variants. Returns the first one 

83 found that isn't in used_aliases. 

84 """ 

85 for char in name: 

86 for transform in [lambda s: s.lower(), lambda s: s.upper()]: 

87 alias = transform(char) 

88 assert len(alias) == 1, f"One character became more than one: `{char}` -> `{alias}`" 

89 if alias not in used_aliases: 

90 return alias 

91 return None

94def get_ancestors(command_name: str) -> List[str]: 

95 """ 

96 List all ancestors for a given command. For example, `foo bar bat` yeilds a list with: 

97 * `__root__` 

98 * `__root__ foo` 

99 * `__root__ foo bar` 

100 

101 Note that `__root__` is always an implicit ancestor. 

102 """ 

103 if command_name == "__root__": 

104 return [] 

105 tokens = command_name.split(" ") 

106 return ["__root__"] + ["".join(tokens[: i + 1]) for i in range(len(tokens))][:-1]

109def normalize_name(name: str, spaces: bool = True) -> str: 

110 """ 

111 Normalizes a name. It's converted to lowercase, leading and trailing `_` are stripped, and `_` is converted to `-`. 

112 If `spaces` is true, it also converts `__` to a single space ` `. 

113 """ 

114 result = name.lower().strip("_") 

115 if spaces: 

116 result = result.replace("__", " ") 

117 result = result.replace("_", "-") 

118 if len(result.strip("- ")) == 0: 

119 raise ArguablyException(f"Cannot normalize name `{name}` - cannot just be underscores and dashes.") 

120 return result

133def get_enum_member_docs(enum_class: Type[enum.Enum]) -> Dict[str, str]: 

134 """Extracts docstrings for enum members similar to PEP-224, which has become a pseudo-standard supported by a lot of 

135 tooling""" 

136 parsed = ast.parse(inspect.getsource(enum_class)) 

137 assert len(parsed.body) == 1 

138 

139 classdef = parsed.body[0] 

140 assert isinstance(classdef, ast.ClassDef) 

141 

142 # Search for a string expression following an assignment 

143 prev = None 

144 result: Dict[str, str] = dict() 

145 for item in classdef.body: 

146 if isinstance(item, ast.Expr) and isinstance(item.value, ast.Constant) and isinstance(item.value.value, str): 

147 if isinstance(prev, ast.Assign) and len(prev.targets) == 1 and isinstance(prev.targets[0], ast.Name): 

148 result[cast(ast.Name, prev.targets[0]).id] = item.value.value 

149 prev = item 

150 

151 return result

154def info_for_flags(cli_arg_name: str, flag_class: Type[enum.Flag]) -> List[EnumFlagInfo]: 

155 """Generates a list of `_EnumFlagInfo` corresponding to all flags in an `enum.Flag`.""" 

156 result = list() 

157 docs = docparse(flag_class.__doc__ or "") 

158 enum_member_docs = get_enum_member_docs(flag_class) 

159 for item in flag_class: 

160 options = [f"--{normalize_name(cast(str, item.name))}"] 

161 arg_description = "" 

162 

163 # `docstring_parser` does not specially parse out attibutes declared in the docstring - we have to do that 

164 # ourselves here. 

165 found = False 

166 for doc_item in docs.meta: 

167 assert len(doc_item.args) >= 2 

168 doc_item_type, doc_item_name = doc_item.args[0], doc_item.args[-1] 

169 if item.name != doc_item_name: 

170 continue 

171 if doc_item_type not in ["var", "cvar", "attribute", "attr"]: 

172 continue 

173 arg_description = doc_item.description or "" 

174 

175 found = True 

176 break 

177 

178 if not found and item.name in enum_member_docs: 

179 # noinspection PyTypeChecker 

180 arg_description = enum_member_docs[item.name] 

181 

182 # Extract the alias from the docstring for the flag item 

183 if alias_match := re.match(r"^\[-([a-zA-Z0-9])] ", arg_description): 

184 arg_description = arg_description[len(alias_match.group(0)) :] 

185 options.insert(0, f"-{alias_match.group(1)}") 

186 

187 result.append( 

188 EnumFlagInfo(cast(Union[Tuple[str], Tuple[str, str]], tuple(options)), cli_arg_name, item, arg_description) 

189 ) 

190 return result

231def run_redirected_io(mp_ctx: multiprocessing.context.SpawnContext, target: Callable, args: Tuple[Any, ...]) -> None: 

232 """Redirects the subprocess's stdout/stderr back to THIS process's stdout/stderr""" 

233 from threading import Thread 

234 

235 # Set up multiprocessing so we can launch a new process to load the file 

236 # We redirect stdout and stderr back to us and print in threads

308def get_callable_methods(cls: type) -> List[Callable]: 

309 """ 

310 Gets all the callable methods from a function - __init__, classmethods, and staticmethods. Skips abstractmethods. 

311 """ 

312 callable_methods = [] 

313 

314 for name, method in vars(cls).items(): 

315 if (name.startswith("__") and name.endswith("__")) or inspect.isabstract(method): 

316 continue 

317 if isinstance(method, staticmethod) or isinstance(method, classmethod): 

318 callable_methods.append(getattr(cls, name)) 

319 

320 return callable_methods

323def load_and_run_inner(file: Path, *args: str, debug: bool) -> LoadAndRunResult: 

324 import arguably 

325 

326 if debug: 

327 logging.basicConfig(level=logging.DEBUG, format="%(pathname)s:%(lineno)d: %(message)s") 

328 

329 # Load the specified file 

330 spec = importlib.util.spec_from_file_location("_arguably_imported", str(file)) 

331 assert spec is not None 

332 assert spec.loader is not None 

333 module = importlib.util.module_from_spec(spec) 

334 sys.modules["_arguably_imported"] = module 

335 spec.loader.exec_module(module) 

336 

337 # Collect all callables (classes and functions) 

338 callables = [item for item in vars(module).values() if callable(item)] 

339 

340 functions = list() 

341 classes = list() 

342 for callable_ in callables: 

343 if isinstance(callable_, type): 

344 classes.append(callable_) 

345 else: 

346 functions.append(callable_) 

347 

348 # For classmethods and staticmethods, we prepend the class name when we call `arguably.command`. Keep track of the 

349 # real name here so we can revert it after 

350 real_names: Dict[Any, str] = dict() 

351 

352 # Add classmethods and staticmethods 

353 for cls in classes: 

354 # Add the initializer for the class itself, if it's not abstract 

355 if not inspect.isabstract(cls): 

356 functions.append(cls) 

357 

358 # Add classmethods and staticmethods 

359 for callable_method in get_callable_methods(cls): 

360 if inspect.ismethod(callable_method): 

361 # We have to set through .__func__ for the bound @classmethod 

362 callable_method = cast(classmethod, callable_method) # type: ignore[assignment] 

363 real_names[callable_method] = callable_method.__name__ 

364 callable_method.__func__.__name__ = f"{cls.__name__}.{callable_method.__name__}" 

365 else: 

366 real_names[callable_method] = callable_method.__name__ 

367 callable_method.__name__ = f"{cls.__name__}.{callable_method.__name__}" 

368 functions.append(callable_method) 

369 

370 arguably._context.context.reset() 

371 

372 # Add all functions to arguably 

373 for function in functions: 

374 try: 

375 # Heuristic for determining what is close enough to a class or function 

376 inspect.signature(function) 

377 get_type_hints(function, include_extras=True) 

378 except TypeError: 

379 continue 

380 

381 try: 

382 arguably.command(function) 

383 except arguably.ArguablyException as e: 

384 warn(f"Unable to add function {function.__name__}: {str(e)}", function) 

385 continue 

386 

387 # If it's a classmethod or staticmethod, revert the name 

388 if function in real_names: 

389 if inspect.ismethod(function): 

390 function.__func__.__name__ = real_names[function] 

391 else: 

392 function.__name__ = real_names[function] 

393 

394 sys.argv.extend(args) 

395 

396 # Run and return success 

397 arguably.run(name=file.stem, always_subcommand=True, show_types=True, show_defaults=True) 

398 return LoadAndRunResult()

401def load_and_run(results: multiprocessing.Queue, file: Path, argv: List[str], debug: bool) -> None: 

402 """Load the specified file, register all callable top-level functions, classmethods, and staticmethods, then run""" 

403 try: 

404 results.put(load_and_run_inner(file, *argv, debug=debug)) 

405 except BaseException as e: 

406 results.put(LoadAndRunResult(exception=e))
