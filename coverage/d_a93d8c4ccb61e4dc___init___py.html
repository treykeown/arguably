226def _split_unquoted(unsplit: str, delimeter: str, limit: Union[int, float] = math.inf) -> List[str]: 

227 """Splits text at a delimiter, as long as that delimiter is not quoted (either single ' or double quotes ").""" 

228 assert len(delimeter) == 1 

229 assert limit > 0 

230 result = list() 

231 quote_char = None 

232 accumulator: List[str] = list() 

233 for char in unsplit: 

234 if char == delimeter and quote_char is None and limit > 0: 

235 result.append("".join(accumulator)) 

236 accumulator.clear() 

237 limit -= 1 

238 continue 

239 elif char == "'": 

240 if quote_char is None: 

241 quote_char = "'" 

242 elif quote_char == "'": 

243 quote_char = None 

244 elif char == '"': 

245 if quote_char is None: 

246 quote_char = '"' 

247 elif quote_char == '"': 

248 quote_char = None 

249 accumulator.append(char) 

250 result.append("".join(accumulator)) 

251 accumulator.clear() 

252 return result

636 def _normalize_type_union(

641 """ 

642 We break this out because Python 3.10 seems to want to wrap `Annotated[Optional[...` in another `Optional`, so

645 if isinstance(value_type, UnionType) or get_origin(value_type) is Union:

647 if len(filtered_types) != 1:

655 @staticmethod 

656 def normalize_type( 

657 function_name: str, 

658 param: inspect.Parameter, 

659 hints: Dict[str, Any], 

660 ) -> Tuple[type, List[_CommandArgModifier]]: 

661 """ 

662 Normalizes the parameter type. Most of the logic here is validation. Explanation of what's returned for a given 

663 parameter type: 

664 * SomeType -> value_type=SomeType, modifiers=[] 

665 * int | None -> value_type=int, modifiers=[] 

666 * Tuple[float, float] -> value_type=type(None), modifiers=[_TupleModifier([float, float])] 

667 * List[str] -> value_type=str, modifiers=[_ListModifier()] 

668 * Annotated[int, arg.count()] -> value_type=int, modifiers=[_CountedModifier()] 

669 

670 Things that will cause an exception: 

671 * Parameterized type other than a Optional[] or Tuple[] 

672 * Flexible-length Tuple[SomeType, ...] 

673 * Parameter lacking an annotation 

674 """ 

675 

676 modifiers: List[_CommandArgModifier] = list() 

677 

678 if param.name in hints: 

679 value_type = hints[param.name] 

680 else: 

681 # No type hint. Guess type from default value, if any other than None. Otherwise, default to string. 

682 value_type = type(param.default) if param.default not in [param.empty, None] else str 

683 

684 # Extra call to normalize a union here, see note in `_normalize_type_union` 

685 value_type = _CommandArg._normalize_type_union(function_name, param, value_type) 

686 

687 # Handle annotated types 

688 if get_origin(value_type) == Annotated: 

689 type_args = get_args(value_type) 

690 if len(type_args) == 0: 

691 raise ArguablyException(f"Function parameter `{param.name}` is Annotated, but no type is specified") 

692 else: 

693 value_type = type_args[0] 

694 for type_arg in type_args[1:]: 

695 if not isinstance(type_arg, _CommandArgModifier): 

696 raise ArguablyException( 

697 f"Function parameter `{param.name}` has an invalid annotation value: {type_arg}" 

698 ) 

699 modifiers.append(type_arg) 

700 

701 # Normalize Union with None 

702 value_type = _CommandArg._normalize_type_union(function_name, param, value_type) 

703 

704 # Validate list/tuple and error on other parameterized types 

705 origin = get_origin(value_type) 

706 if origin == list: 

707 type_args = get_args(value_type) 

708 if len(type_args) == 0: 

709 value_type = str 

710 elif len(type_args) > 1: 

711 raise ArguablyException( 

712 f"Function parameter `{param.name}` in `{function_name}` has too many items passed to List[...]." 

713 f"There should be exactly one item between the square brackets." 

714 ) 

715 else: 

716 value_type = type_args[0] 

717 modifiers.append(_ListModifier()) 

718 elif origin == tuple: 

719 if param.kind in [param.VAR_KEYWORD, param.VAR_POSITIONAL]: 

720 raise ArguablyException( 

721 f"Function parameter `{param.name}` in `{function_name}` is an *args or **kwargs, which should " 

722 f"be annotated with what only one of its items should be." 

723 ) 

724 type_args = get_args(value_type) 

725 if len(type_args) == 0: 

726 raise ArguablyException( 

727 f"Function parameter `{param.name}` in `{function_name}` is a tuple but doesn't specify the " 

728 f"type of its items, which arguably requires." 

729 ) 

730 if type_args[-1] is Ellipsis: 

731 raise ArguablyException( 

732 f"Function parameter `{param.name}` in `{function_name}` is a variable-length tuple, which is " 

733 f"not supported." 

734 ) 

735 value_type = type(None) 

736 modifiers.append(_TupleModifier(list(type_args))) 

737 elif origin is not None: 

738 if param.kind in [param.VAR_KEYWORD, param.VAR_POSITIONAL]: 

739 raise ArguablyException( 

740 f"Function parameter `{param.name}` in `{function_name}` is an *args or **kwargs, which should " 

741 f"be annotated with what only one of its items should be." 

742 ) 

743 raise ArguablyException( 

744 f"Function parameter `{param.name}` in `{function_name}` is a generic type " 

745 f"(`{get_origin(value_type)}`), which is not supported." 

746 ) 

747 

748 return value_type, modifiers

775 def call(self, parsed_args: Dict[str, Any]) -> Any: 

776 """Filters arguments from argparse to only include the ones used by this command, then calls it""" 

777 

778 args = list() 

779 kwargs = dict() 

780 

781 filtered_args = dict() 

782 for k, v in parsed_args.items(): 

783 if k in self.arg_map: 

784 filtered_args[self.arg_map[k].param_name] = v 

785 param_to_arg_name = {self.arg_map[k].param_name: k for k in self.arg_map} 

786 

787 for param_name, param in inspect.signature(self.function).parameters.items(): 

788 arg_value = filtered_args[param_name] 

789 

790 # Build the appropriate class from any given `_BuildTypeSpec`s 

791 if isinstance(arg_value, _BuildTypeSpec): 

792 arg_value_type = self.arg_map[param_name].arg_value_type 

793 arg_value = _context.resolve_subtype(arg_value_type, arg_value, param_name) 

794 

795 # Add to either args or kwargs 

796 if param.kind in [param.POSITIONAL_ONLY, param.POSITIONAL_OR_KEYWORD]: 

797 args.append(arg_value) 

798 elif param.kind == param.VAR_POSITIONAL: 

799 args.extend(arg_value) 

800 else: 

801 kwargs[param_to_arg_name[param_name]] = arg_value 

802 

803 # Call the function 

804 return self.function(*args, **kwargs)

1053 def _process_decorator_info(self, info: _CommandDecoratorInfo) -> _Command: 

1054 """Takes the decorator info and return a processed command""" 

1055 

1056 processed_name = info.name 

1057 

1058 # Get the description from the docstring 

1059 if info.function.__doc__ is None: 

1060 docs = None 

1061 processed_description = "" 

1062 else: 

1063 docs = docparse(info.function.__doc__) 

1064 processed_description = "" if docs.short_description is None else docs.short_description 

1065 

1066 hints = get_type_hints(info.function, include_extras=True) 

1067 

1068 # Will be filled in as we loop over all parameters 

1069 processed_args: List[_CommandArg] = list() 

1070 used_arg_aliases: List[str] = list() 

1071 has_positional_args = False 

1072 variadic_positional_arg = None 

1073 

1074 # Iterate over all parameters 

1075 for param_name, param in inspect.signature(info.function).parameters.items(): 

1076 arg_name = _normalize_name(param_name, spaces=False) 

1077 arg_default = _NoDefault if param.default is param.empty else param.default 

1078 

1079 # Handle variadic arguments 

1080 arg_count = 1 

1081 if param.kind is param.VAR_KEYWORD: 

1082 raise ArguablyException(f"`{processed_name}` is using **kwargs, which is not supported") 

1083 if param.kind is param.VAR_POSITIONAL: 

1084 arg_count = _CommandArg.ANY_COUNT 

1085 variadic_positional_arg = param_name 

1086 

1087 # Get the type and normalize it 

1088 arg_value_type, modifiers = _CommandArg.normalize_type(processed_name, param, hints) 

1089 tuple_modifiers = [m for m in modifiers if isinstance(m, _TupleModifier)] 

1090 if len(tuple_modifiers) > 0: 

1091 assert len(tuple_modifiers) == 1 

1092 arg_count = len(tuple_modifiers[0].tuple_arg) 

1093 

1094 # Get the description 

1095 arg_description = "" 

1096 if docs is not None and docs.params is not None: 

1097 ds_matches = [ds_p for ds_p in docs.params if ds_p.arg_name == param.name] 

1098 if len(ds_matches) > 1: 

1099 raise ArguablyException( 

1100 f"Multiple docstring entries for parameter `{param.name}` in " f"`{processed_name}`" 

1101 ) 

1102 if len(ds_matches) == 1: 

1103 ds_info = ds_matches[0] 

1104 arg_description = "" if ds_info.description is None else ds_info.description 

1105 

1106 # Extract the alias 

1107 arg_alias = None 

1108 if alias_match := re.match(r"^\[-([a-zA-Z0-9])] ", arg_description): 

1109 arg_description = arg_description[len(alias_match.group(0)) :] 

1110 arg_alias = alias_match.group(1) 

1111 

1112 # Generate the alias if required and there isn't one 

1113 if self._options.auto_alias_params and arg_alias is None: 

1114 arg_alias = _find_alias(used_arg_aliases, param.name) 

1115 if arg_alias is not None: 

1116 used_arg_aliases.append(arg_alias) 

1117 

1118 # Extract the metavars 

1119 metavars = None 

1120 if metavar_split := re.split(r"\{((?:[a-zA-Z0-9_-]+(?:, *)*)+)}", arg_description): 

1121 if len(metavar_split) == 3: 

1122 # format would be: ['pre-metavar', 'METAVAR', 'post-metavar'] 

1123 match_items = [i.strip() for i in metavar_split[1].split(",")] 

1124 if arg_count == _CommandArg.ANY_COUNT: 

1125 if len(match_items) != 1: 

1126 raise ArguablyException( 

1127 f"Function parameter `{param.name}` in `{processed_name}` should only have one item in " 

1128 f"its metavar descriptor, but found {len(match_items)}: {','.join(match_items)}." 

1129 ) 

1130 elif len(match_items) != arg_count: 

1131 raise ArguablyException( 

1132 f"Function parameter `{param.name}` in `{processed_name}` takes {arg_count} items, but " 

1133 f"metavar descriptor has {len(match_items)}: {','.join(match_items)}." 

1134 ) 

1135 metavars = [i.upper() for i in match_items] 

1136 arg_description = "".join(metavar_split) # Strips { and } from metavars for description 

1137 if len(metavar_split) > 3: 

1138 raise ArguablyException( 

1139 f"Function parameter `{param.name}` in `{processed_name}` has multiple metavar sequences - " 

1140 f"these are denoted like {{A, B, C}}. There should be only one."

1155 modifier.check_valid(arg_value_type, param, processed_name)

1158 processed_args.append( 

1159 _CommandArg( 

1160 param_name, 

1161 arg_name, 

1162 input_method, 

1163 arg_value_type, 

1164 arg_description, 

1165 arg_count, 

1166 arg_alias,

1169 modifiers, 

1170 ) 

1171 )

1208 def _set_up_args(self, cmd: _Command) -> None: 

1209 """Adds all arguments to the parser for a given command""" 

1210 

1211 parser = self._parsers[cmd.name] 

1212 

1213 for arg_ in cmd.args: 

1214 if arg_.input_method.is_positional: 

1215 if arg_.param_name == self._options.command_metavar: 

1216 raise ArguablyException( 

1217 f"Function argument `{arg_.param_name}` in `{cmd.name}` is named the same as `command_metavar`." 

1218 f" Either change the parameter name or set the `command_metavar` option to something other than" 

1219 f' "{arg_.param_name}" when calling arguably.run()' 

1220 ) 

1221 # Short-circuit, different path for enum.Flag. We add multiple options, one for each flag entry 

1222 if issubclass(arg_.arg_value_type, enum.Flag): 

1223 if arg_.input_method.is_positional: 

1224 raise ArguablyException( 

1225 f"Function argument `{arg_.param_name}` in `{cmd.name}` is both positional and an enum.Flag. " 

1226 f"Positional enum flags are unsupported, since they are turned into options." 

1227 ) 

1228 if arg_.default is _NoDefault: 

1229 raise ArguablyException( 

1230 f"Function argument `{arg_.param_name}` in `{cmd.name}` is an enum.Flag. Due to implementation " 

1231 f"limitations, all enum.Flag parameters must have a default value." 

1232 ) 

1233 parser.set_defaults(**{arg_.arg_name: arg_.default}) 

1234 for entry in _info_for_flags(arg_.arg_name, arg_.arg_value_type): 

1235 parser.add_argument( 

1236 *entry.option, action=_EnumFlagAction, const=entry, nargs=0, help=entry.description 

1237 ) 

1238 continue 

1239 

1240 # Optional kwargs for parser.add_argument 

1241 add_arg_kwargs: Dict[str, Any] = dict(type=arg_.arg_value_type) 

1242 

1243 arg_description = arg_.description 

1244 

1245 # `default` value? 

1246 if arg_.input_method.is_optional and arg_.default is not _NoDefault: 

1247 add_arg_kwargs.update(default=arg_.default) 

1248 if self._options.show_defaults: 

1249 if len(arg_description) > 0: 

1250 arg_description += " " 

1251 if isinstance(arg_.default, enum.Enum): 

1252 arg_description += f"(default: {_normalize_name(arg_.default.name, spaces=False)})" 

1253 elif isinstance(arg_.default, str): 

1254 str_default = arg_.default 

1255 # Use the string repr if it contains spaces, contains a newline, or is zero-length 

1256 if (" " in str_default) or ("\n" in str_default) or (len(str_default) == 0): 

1257 str_default = repr(str_default) 

1258 arg_description += f"(default: {str_default})" 

1259 else: 

1260 arg_description += f"(default: {arg_.default})" 

1261 

1262 # Number of arguments `nargs`? 

1263 if arg_.count is _CommandArg.ANY_COUNT: 

1264 add_arg_kwargs.update(nargs="*") 

1265 elif arg_.input_method is _InputMethod.OPTIONAL_POSITIONAL: 

1266 add_arg_kwargs.update(nargs="?") 

1267 elif arg_.count != 1: 

1268 add_arg_kwargs.update(nargs=arg_.count, action=_CommaSeparatedTupleAction) 

1269 

1270 # Any specified `metavar`s? 

1271 if arg_.metavars is not None: 

1272 if len(arg_.metavars) == 1: 

1273 add_arg_kwargs.update(metavar=arg_.metavars[0]) 

1274 else: 

1275 add_arg_kwargs.update(metavar=tuple(arg_.metavars)) 

1276 

1277 # Possible choices `choices`? 

1278 if issubclass(arg_.arg_value_type, enum.Enum): 

1279 mapping = self.set_up_enum(arg_.arg_value_type) 

1280 add_arg_kwargs.update(choices=[n for n in mapping]) 

1281 

1282 arg_names: Tuple[str, ...] = (arg_.arg_name,) 

1283 

1284 # Special handling for optional arguments 

1285 if arg_.input_method is _InputMethod.OPTION: 

1286 arg_names = (f"--{arg_.arg_name}",) if arg_.alias is None else (f"-{arg_.alias}", f"--{arg_.arg_name}") 

1287 

1288 # `bool` should be flags 

1289 if issubclass(arg_.arg_value_type, bool): 

1290 if arg_.input_method is not _InputMethod.OPTION or arg_.default is _NoDefault: 

1291 raise ArguablyException( 

1292 f"Function argument `{arg_.param_name}` in `{cmd.name}` is a `bool`. Boolean parameters must " 

1293 f"have a default value and be an optional, not a positional, argument." 

1294 ) 

1295 # Use `store_true` or `store_false` for bools 

1296 add_arg_kwargs.update(action="store_true" if arg_.default is False else "store_false") 

1297 if "type" in add_arg_kwargs: 

1298 del add_arg_kwargs["type"] 

1299 

1300 # Set the help description 

1301 add_arg_kwargs.update(help=arg_description) 

1302 

1303 # Run modifiers for this arg 

1304 for modifier in arg_.modifiers: 

1305 modifier.modify_arg_dict(cmd, arg_, add_arg_kwargs) 

1306 

1307 # Add the argument to the parser 

1308 parser.add_argument(*arg_names, **add_arg_kwargs)

1310 def _build_subparser_tree(self, command_decorator_info: _CommandDecoratorInfo) -> str: 

1311 """Builds up the subparser tree for a given `_CommandDecoratorInfo`. Inserts dummy entries to `self._parsers` 

1312 and `self._commands` if necessary. Returns the name of the parent for this command.""" 

1313 

1314 prev_ancestor = "__root__" 

1315 

1316 # Create tree of parsers and subparsers for ancestors 

1317 ancestor_names = _get_ancestors(command_decorator_info.name) 

1318 for ancestor in ancestor_names: 

1319 required_subparser = False 

1320 if ancestor not in self._commands: 

1321 # Dummy command - this ancestor doesn't have a function of its own, it's just a path. 

1322 self._commands[ancestor] = _Command(lambda *_, **__: None, ancestor) 

1323 if ancestor not in self._parsers: 

1324 # Dummy parser - since there's nothing to run, require the subparser. 

1325 required_subparser = True 

1326 self._parsers[ancestor] = self._subparsers[prev_ancestor].add_parser( 

1327 ancestor.split(" ")[-1], help="", formatter_class=self._formatter, **self._extra_argparser_options 

1328 ) 

1329 if ancestor not in self._subparsers: 

1330 # Add subparser to the parent command's parser. 

1331 ancestor_cmd = self._commands[ancestor] 

1332 if ancestor_cmd.has_positional_args: 

1333 raise ArguablyException( 

1334 f"Command `{ancestor}` cannot have both subcommands and positional arguments." 

1335 ) 

1336 self._subparsers[ancestor] = self._parsers[ancestor].add_subparsers( 

1337 parser_class=_ArgumentParser, 

1338 dest=ancestor_cmd.get_subcommand_metavar(self._options.command_metavar), 

1339 metavar=self._options.command_metavar, 

1340 required=required_subparser, 

1341 ) 

1342 prev_ancestor = ancestor 

1343 return prev_ancestor

1354 def run( 

1355 self, 

1356 name: Optional[str] = None, 

1357 call_ancestors: bool = False, 

1358 auto_alias_cmds: bool = False, 

1359 auto_alias_params: bool = False, 

1360 version_flag: Union[bool, Tuple[str], Tuple[str, str]] = False, 

1361 show_defaults: bool = True, 

1362 max_description_offset: int = 60, 

1363 max_width: int = 120, 

1364 command_metavar: str = "command", 

1365 output: Optional[TextIO] = None, 

1366 ) -> Any: 

1367 """Set up the argument parser, parse argv, and run the appropriate command(s)""" 

1368 

1369 # Set options 

1370 self._options = _ContextOptions(**{k: v for k, v in locals().items() if k != "self"}) 

1371 self._extra_argparser_options = dict(output=self._options.output) 

1372 

1373 self._is_calling_target = False 

1374 

1375 # Grab the description 

1376 import __main__ 

1377 

1378 description = "" if __main__.__doc__ is None else __main__.__doc__.partition("\n\n\n")[0] 

1379 

1380 # Set up the root parser 

1381 root_parser = _ArgumentParser( 

1382 prog=self._options.name, 

1383 description=description, 

1384 formatter_class=self._formatter, 

1385 **self._extra_argparser_options, 

1386 ) 

1387 self._parsers["__root__"] = root_parser 

1388 

1389 # Add version flags if necessary 

1390 argparse_version_flags: Union[tuple, Tuple[str], Tuple[str, str]] = tuple() 

1391 if self._options.version_flag: 

1392 if not hasattr(__main__, "__version__"): 

1393 raise ArguablyException("__version__ must be defined if version_flag is set") 

1394 if isinstance(self._options.version_flag, tuple): 

1395 argparse_version_flags = self._options.version_flag 

1396 else: 

1397 argparse_version_flags = ("--version",) 

1398 root_parser.add_argument( 

1399 *argparse_version_flags, action="version", version=f"%(prog)s {__main__.__version__}" 

1400 ) 

1401 

1402 # Check the number of commands we have 

1403 only_one_cmd = len(self._command_decorator_info) == 1 

1404 if len(self._command_decorator_info) == 0: 

1405 raise ArguablyException("At least one command is required") 

1406 

1407 for command_decorator_info in sorted( 

1408 self._command_decorator_info, key=lambda v: (v.name != "__root__", v.name.count(" ")) 

1409 ): 

1410 if only_one_cmd: 

1411 parent_name = "__root__" 

1412 self._parsers[command_decorator_info.name] = self._parsers["__root__"] 

1413 else: 

1414 parent_name = self._build_subparser_tree(command_decorator_info) 

1415 

1416 # Assign an alias, if needed 

1417 if command_decorator_info.alias is None and self._options.auto_alias_cmds: 

1418 command_decorator_info.alias = _find_alias(self._command_aliases.keys(), command_decorator_info.name) 

1419 

1420 # Process command and its args 

1421 cmd = self._process_decorator_info(command_decorator_info) 

1422 

1423 # Save command and its alias to the dicts 

1424 if cmd.name in self._commands: 

1425 raise ArguablyException(f"Name `{cmd.name}` is already taken") 

1426 if cmd.alias is not None: 

1427 if cmd.alias in self._command_aliases: 

1428 raise ArguablyException( 

1429 f"Alias `{cmd.alias}` for `{cmd.name}` is already taken by " 

1430 f"`{self._command_aliases[cmd.alias]}`" 

1431 ) 

1432 self._command_aliases[cmd.alias] = cmd.name 

1433 self._commands[cmd.name] = cmd 

1434 

1435 # Add the parser for the command 

1436 if not only_one_cmd and cmd.name != "__root__": 

1437 self._parsers[cmd.name] = self._subparsers[parent_name].add_parser( 

1438 cmd.name.split(" ")[-1], 

1439 aliases=[cmd.alias] if cmd.alias is not None else [], 

1440 help=cmd.description, 

1441 description=cmd.description, 

1442 formatter_class=self._formatter, 

1443 **self._extra_argparser_options, 

1444 ) 

1445 

1446 # Add the arguments to the command's parser 

1447 try: 

1448 self._set_up_args(cmd) 

1449 except argparse.ArgumentError as e: 

1450 # Special handling for version flags for __root__ 

1451 if cmd.name != "__root__": 

1452 raise e 

1453 if not e.message.startswith("conflicting option string"): 

1454 raise e 

1455 # e.message == `conflicting option strings: -v, --version` 

1456 conflicts = e.message.split(":")[1].strip().split(", ") 

1457 filtered_conflicts = [c for c in conflicts if c in argparse_version_flags] 

1458 if len(filtered_conflicts) == 0: 

1459 raise e 

1460 raise ArguablyException( 

1461 f"Conflict due to `version_flag` being set and __root__ having a parameter with a conflicting name." 

1462 f" Conflicting args: {', '.join(conflicts)}" 

1463 ) 

1464 

1465 # Use the function description, not the __main__ docstring, if only one command 

1466 if only_one_cmd: 

1467 self._parsers["__root__"].description = next(iter(self._commands.values())).description 

1468 

1469 # Make the magic happen 

1470 parsed_args = vars(root_parser.parse_args()) 

1471 

1472 # Resolve the command that needs to be called 

1473 if only_one_cmd: 

1474 cmd = next(iter(self._commands.values())) 

1475 self._current_parser = self._parsers["__root__"] 

1476 else: 

1477 # Find the actual command we need to execute by traversing the subparser tree. Call each stop along the way 

1478 # if the call_ancestors option is set to True. 

1479 path = "__root__" 

1480 while path in self._subparsers: 

1481 # Find the variable name for this subparser's command metavar and read the value. If it's none, run the 

1482 # current stop of our path in the tree. 

1483 subcmd_metavar = self._commands[path].get_subcommand_metavar(self._options.command_metavar) 

1484 subcmd_name = parsed_args[subcmd_metavar] 

1485 if subcmd_name is None: 

1486 break 

1487 

1488 # Resolve any command aliases 

1489 if subcmd_name in self._command_aliases: 

1490 subcmd_name = self._command_aliases[subcmd_name] 

1491 

1492 # Call the ancestor if call_ancestors is set 

1493 if self._options.call_ancestors: 

1494 self._is_calling_target = False 

1495 self._current_parser = self._parsers[path] 

1496 self._commands[path].call(parsed_args) 

1497 

1498 # Update the path and continue 

1499 if path == "__root__": 

1500 path = subcmd_name 

1501 else: 

1502 path += f" {subcmd_name}" 

1503 

1504 # If the command is unknown, print the help for the most recent parent 

1505 if path not in self._commands: 

1506 self._parsers[path.rsplit(" ", 1)[0]].print_help(file=self._options.output) 

1507 return None 

1508 if path == "__root__": 

1509 root_parser.print_help(file=self._options.output) 

1510 return None 

1511 

1512 # Found command 

1513 self._current_parser = self._parsers[path] 

1514 cmd = self._commands[path] 

1515 

1516 self._is_calling_target = True 

1517 result = cmd.call(parsed_args) 

1518 self._current_parser = None 

1519 return result

1521 def _build_subtype( 

1522 self, subtype_info: _SubtypeDecoratorInfo, type_spec: _BuildTypeSpec, parent_param_name: str 

1523 ) -> Any: 

1524 type_ = subtype_info.type_ 

1525 factory = subtype_info.factory or type_.__call__ 

1526 template = subtype_info.factory or type_.__init__ # type: ignore[misc] 

1527 hints = get_type_hints(template, include_extras=True) 

1528 normalized_kwargs: Dict[str, Any] = dict() 

1529 

1530 missing_required_keys = [ 

1531 _normalize_name(p) 

1532 for p in inspect.signature(template).parameters 

1533 if p not in type_spec.kwargs and p != "self" 

1534 ] 

1535 if len(missing_required_keys) > 0: 

1536 params = inspect.signature(template).parameters 

1537 missing_specs = list() 

1538 for key in missing_required_keys: 

1539 arg_value_type, modifiers = _CommandArg.normalize_type(type_.__name__, params[key], hints) 

1540 missing_specs.append(f"{key} ({arg_value_type.__name__})") 

1541 self.error(f"the following keys are required for {parent_param_name}: {', '.join(missing_specs)}") 

1542 

1543 # Iterate over all parameters 

1544 for param_name, param in inspect.signature(template).parameters.items(): 

1545 try: 

1546 arg_name = _normalize_name(param_name) 

1547 if arg_name == "self": 

1548 continue 

1549 param_value = type_spec.kwargs[arg_name] 

1550 del type_spec.kwargs[arg_name] 

1551 arg_value_type, modifiers = _CommandArg.normalize_type(type_.__name__, param, hints) 

1552 except ArguablyException: 

1553 raise ArguablyException(f"Error processing parameter {param_name} of subtype {type_.__name__}") 

1554 if len(modifiers) > 0: 

1555 raise ArguablyException( 

1556 f"Error processing parameter {param_name} of subtype {type_.__name__}: Cannot use modifiers " 

1557 f"on subtypes" 

1558 ) 

1559 normalized_kwargs[param_name] = arg_value_type(param_value) 

1560 

1561 # The calls to .error() cause an exit 

1562 if len(type_spec.kwargs) > 1: 

1563 self.error(f"unexpected keys for {parent_param_name}: {', '.join(type_spec.kwargs)}") 

1564 elif len(type_spec.kwargs) > 0: 

1565 self.error(f"unexpected key for {parent_param_name}: {next(iter(type_spec.kwargs))}") 

1566 

1567 return factory(**normalized_kwargs)

1569 def resolve_subtype(self, arg_value_type: type, type_spec: _BuildTypeSpec, param_name: str) -> Any: 

1570 options = self.find_subtype(arg_value_type) 

1571 if len(options) == 0: 

1572 options = [_SubtypeDecoratorInfo(arg_value_type)] 

1573 if len(options) == 1: 

1574 return self._build_subtype(options[0], type_spec, param_name) 

1575 matches = [op for op in options if op.alias == type_spec.subtype] 

1576 if len(matches) == 0: 

1577 self.error(f"unknown subtype `{type_spec.subtype}` for {param_name}") 

1578 if len(matches) > 1: 

1579 raise ArguablyException(f"More than one match for subtype `{type_spec.subtype}` of type {arg_value_type}") 

1580 return self._build_subtype(matches[0], type_spec, param_name)
